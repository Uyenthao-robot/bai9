Luyện tập

Sửa lại chương trình trong Hình 9.5a, sử dụng thuộc tính Style:

Họ tên Toán Vật lí Hóa học
Nguyễn Văn A 8.5 7.0 9.0
Trần Thị B 7.0 8.0 8.5

Vận dụng

Tạo bảng HTML:

def generate_html_table(data):
html_content = "<table>"
html_content += "<tr>"
html_content += "<th>Họ tên</th><th>Điểm Toán</th><th>Điểm Vật lí</th><th>Điểm Hóa học</th>"
html_content += "</tr>"
for row in data:
html_content += "<tr>"
for value in row:
html_content += f"<td>{value}</td>"
html_content += "</tr>"
html_content += "</table>"
return html_content

# Dữ liệu đầu vào
data = [
["Nguyễn Văn A", "8.5", "7.0", "9.0"],
["Trần Thị B", "7.0", "8.0", "8.5"],
["Lê Quang C", "6.5", "7.5", "8.0"],
["Phạm Thị D", "9.0", "8.5", "7.5"]
]

# Xuất file
html_content = generate_html_table(data)
with open("table.html", "w") as file:
file.write(html_content)
print("HTML table created!")
